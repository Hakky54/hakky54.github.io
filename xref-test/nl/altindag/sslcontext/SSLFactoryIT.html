12  import static nl.altindag.sslcontext.TestConstants.KEYSTORE_LOCATION;
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  class SSLFactoryIT {
16  
17      private static final Logger LOGGER = LoggerFactory.getLogger(SSLFactoryIT.class);
18  
19      @Test
20      void executeHttpsRequestWithMutualAuthentication() throws IOException {
21          LogCaptor logCaptor = LogCaptor.forRoot();
22  
23          SSLFactory sslFactory = SSLFactory.builder()
24                  .withIdentityMaterial(KEYSTORE_LOCATION + "badssl-identity.p12", "badssl.com".toCharArray())
25                  .withTrustMaterial(KEYSTORE_LOCATION + "badssl-truststore.p12", "badssl.com".toCharArray())
26                  .build();
27  
28          HttpsURLConnection connection = (HttpsURLConnection) new URL("https://client.badssl.com/").openConnection();
29          connection.setSSLSocketFactory(sslFactory.getSslContext().getSocketFactory());
30          connection.setHostnameVerifier(sslFactory.getHostnameVerifier());
31          connection.setRequestMethod("GET");
32  
33          int statusCode = connection.getResponseCode();
34  
35          if (statusCode == 400) {
36              LOGGER.warn("Certificate may have expired and needs to be updated");
37          } else {
38              assertThat(statusCode).isEqualTo(200);
39              assertThat(logCaptor.getLogs()).containsExactly("Received the following server certificate: [CN=*.badssl.com, O=Lucas Garron Torres, L=Walnut Creek, ST=California, C=US]");
40          }
41      }

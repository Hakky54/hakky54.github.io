18  /**
19   * {@link CompositeX509ExtendedTrustManager} is a wrapper for a collection of TrustManagers.
20   * It has the ability to validate a certificate chain against multiple TrustManagers.
21   * If any one of the composed managers trusts a certificate chain, then it is trusted by the composite manager.
22   * The TrustManager can be build from one or more of any combination provided within the {@link Builder CompositeX509ExtendedTrustManager.Builder}.
23   * <br><br>
24   * This includes:
25   * <pre>
26   *     - Any amount of custom TrustManagers
27   *     - Any amount of custom TrustStores
28   * </pre>
29   *
30   * @see <a href="http://stackoverflow.com/questions/1793979/registering-multiple-keystores-in-jvm">
31   *     http://stackoverflow.com/questions/1793979/registering-multiple-keystores-in-jvm
32   *     </a>
33   * @see <a href="http://codyaray.com/2013/04/java-ssl-with-multiple-keystores">
34   *     http://codyaray.com/2013/04/java-ssl-with-multiple-keystores
35   *     </a>
36   */
37  public final class CompositeX509ExtendedTrustManager extends X509ExtendedTrustManager {
38  
39      private static final Logger LOGGER = LoggerFactory.getLogger(CompositeX509ExtendedTrustManager.class);
40      private static final String CERTIFICATE_EXCEPTION_MESSAGE = "None of the TrustManagers trust this certificate chain";
41      private static final String CLIENT_CERTIFICATE_LOG_MESSAGE = "Received the following client certificate: [{}]";
42      private static final String SERVER_CERTIFICATE_LOG_MESSAGE = "Received the following server certificate: [{}]";
43  
44      private final List<? extends X509ExtendedTrustManager> trustManagers;
45      private final X509Certificate[] acceptedIssuers;
46  
47      public CompositeX509ExtendedTrustManager(List<? extends X509ExtendedTrustManager> trustManagers) {
48          this.trustManagers = Collections.unmodifiableList(trustManagers);
49          acceptedIssuers = trustManagers.stream()
50                  .map(X509ExtendedTrustManager::getAcceptedIssuers)
51                  .flatMap(Arrays::stream)
52                  .distinct()
53                  .toArray(X509Certificate[]::new);
54      }
55  
56      @Override
57      public void checkClientTrusted(X509Certificate[] chain, String authType) throws CertificateException {
58          if (LOGGER.isDebugEnabled()) {
59              LOGGER.debug(CLIENT_CERTIFICATE_LOG_MESSAGE, chain[0].getSubjectDN());
60          }
61  
62          List<CertificateException> certificateExceptions = new ArrayList<>();
63          for (X509ExtendedTrustManager trustManager : trustManagers) {
64              try {
65                  trustManager.checkClientTrusted(chain, authType);
66                  return;
67              } catch (CertificateException e) {
68                  certificateExceptions.add(e);
69              }
70          }
71  
72          CertificateException certificateException = new CertificateException(CERTIFICATE_EXCEPTION_MESSAGE);
73          certificateExceptions.forEach(certificateException::addSuppressed);
74  
75          throw certificateException;
76      }
77

201     public static final class Builder {
202 
203         private final List<X509ExtendedTrustManager> trustManagers = new ArrayList<>();
204 
205         public <T extends X509ExtendedTrustManager> Builder withTrustManagers(T... trustManagers) {
206             return withTrustManagers(Arrays.asList(trustManagers));
207         }
208 
209         public Builder withTrustManagers(List<? extends X509ExtendedTrustManager> trustManagers) {
210             this.trustManagers.addAll(trustManagers);
211             return this;
212         }
213 
214         public <T extends KeyStore> Builder withTrustStores(T... trustStores) {
215             return withTrustStores(Arrays.asList(trustStores));
216         }
217 
218         public Builder withTrustStores(List<? extends KeyStore> trustStores) {
219             for (KeyStore trustStore : trustStores) {
220                 this.trustManagers.add(TrustManagerUtils.createTrustManager(trustStore));
221             }
222             return this;
223         }
224 
225         public <T extends KeyStore> Builder withTrustStore(T trustStore) {
226             this.trustManagers.add(TrustManagerUtils.createTrustManager(trustStore));
227             return this;
228         }
229 
230         public <T extends KeyStore> Builder withTrustStore(T trustStore, String trustManagerAlgorithm) {
231             this.trustManagers.add(TrustManagerUtils.createTrustManager(trustStore, trustManagerAlgorithm));
232             return this;
233         }
234 
235         public CompositeX509ExtendedTrustManager build() {
236             return new CompositeX509ExtendedTrustManager(trustManagers);
237         }
